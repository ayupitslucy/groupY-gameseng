project(setup)
cmake_minimum_required(VERSION 3.11)
# Require modern C++
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

#### Setup Directories ####
# Main output directory
SET(OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin/")
# Output all DLLs from all libs into main build folder
SET(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${OUTPUT_DIRECTORY})

#### Add External Dependencies ####
add_subdirectory("lib/SFML")
set(SFML_INCS "lib/SFML/include")
link_directories("${CMAKE_BINARY_DIR}/lib/SFML/lib")

#### Engine ####
add_library(engine STATIC engine/game_system.cpp engine/scene.cpp engine/entity_manager.cpp engine/entity_component_system.cpp engine/renderer.cpp engine/tile_engine.cpp "src/sound.cpp" "src/tower.cpp" "src/tower.h" "src/enemy.h" "src/enemy.cpp" "src/projectile.cpp" "src/projectile.h"  "src/game_entities.cpp")
target_include_directories(engine PUBLIC engine)
target_link_libraries(engine sfml-graphics sfml-audio)

add_custom_target(copy_resources ALL COMMAND ${CMAKE_COMMAND}
  -E copy_directory
    "${PROJECT_SOURCE_DIR}/res"
    ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}/res
)

#### Copy OpenAL DLL ####
SET(OPENAL_DLL_PATH "${PROJECT_SOURCE_DIR}/lib/SFML/extlibs/bin/x64/openal32.dll")
add_custom_target(copy_openal_dll ALL COMMAND ${CMAKE_COMMAND}
  -E copy
    ${OPENAL_DLL_PATH}
    ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}
)

#### Tower Defence Game
add_executable(td_game src/main.cpp src/scenes.cpp src/components.cpp "src/sound.cpp" "src/tower.cpp" "src/tower.h" "src/enemy.h" "src/enemy.cpp" "src/projectile.cpp" "src/projectile.h"  "src/game_entities.cpp")
target_include_directories(td_game PRIVATE ${SFML_INCS} .)
target_link_libraries(td_game engine sfml-graphics sfml-audio)
add_dependencies(td_game copy_resources)
add_dependencies(td_game copy_openal_dll)
